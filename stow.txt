sudo apt-get install stow    # To install gnu stow on ubuntu
mkdir -p ~/dotfiles/{bash,git,nvim,tmux} && cd dotfiles     # Create a stow dir (i believe it can be any where)
mv ~/.bashrc ~/dotfiles/bash/.bashrc   # copying each app config to its .dotfiles subfolder

to stow the config
cd ~/dotfiles
stow bash       # creates symlink ~/.bashrc → ~/dotfiles/bash/.bashrc
stow git        # symlinks ~/.gitconfig → ~/dotfiles/git/.gitconfig
stow bash nvim tmux   # deploy all three in one command
stow -D bash    # removes only the symlinks that were created for 'bash'
stow -d ~/dotfiles -t ~/.config nvim   # -d: specify the stow directory  -t: specify the target directory


symlinks
Syntax: ln -s TARGET LINK_NAME
ln -s /etc/myapp/config ~/myconfig    # link 'myconfig' points to '/etc/myapp/config'
# Remove a symlink (not the target!)
rm ~/myconfig
# or:
unlink ~/myconfig
